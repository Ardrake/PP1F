            do
            {
                if (SelectionTransaction.selectedTransac == "Retrait" && montantRetrait > 0 && montantRetrait <= 1000)
                {
                    if (montantRetrait % 10 != 0)
                    {
                        MessageBox.Show("Le montant doit etre un multiple de 10");
                    }
                    else
                    {
                        if (montantRetrait > 0 || montantRetrait < 1000)
                        {
                            MessageBox.Show("Le montant de retrait doit etre entre 0$ et 1000$");
                        }
                        else
                        {
                            if (DataGuichet.totalGuichet > montantRetrait)
                            {
                                if (SelectionCompte.selectedCompteString == "Cheque")
                                {
                                    // Retrait dans Cheque
                                    if (ClientCourant.compteCheque.Balance >= montantRetrait)
                                    {
                                        ClientCourant.compteCheque.Retrait(montantRetrait);
                                        laNouvelleBalance = ClientCourant.compteCheque.Balance;
                                        DataGuichet.totalGuichet -= (int)montantRetrait;
                                        TransacReussi = true;
                                    }
                                    else
                                        fondInsufisant = true;
                                }
                                if (SelectionCompte.selectedCompteString == "Épargne")
                                {
                                    // Retrait dans Epargne
                                    if (ClientCourant.compteEpargne.Balance >= montantRetrait)
                                    {
                                        ClientCourant.compteEpargne.Retrait(montantRetrait);
                                        laNouvelleBalance = ClientCourant.compteEpargne.Balance;
                                        DataGuichet.totalGuichet -= (int)montantRetrait;
                                        TransacReussi = true;
                                    }
                                    else
                                        fondInsufisant = true;
                                }
                                if (SelectionTransaction.selectedTransac == "Dépot" && montantRetrait > 0)
                                {
                                    if (SelectionCompte.selectedCompteString == "Cheque")
                                    {
                                        ClientCourant.compteCheque.Depot(montantRetrait);
                                        laNouvelleBalance = ClientCourant.compteCheque.Balance;
                                        TransacReussi = true;
                                    }
                                    if (SelectionCompte.selectedCompteString == "Épargne")
                                    {
                                        ClientCourant.compteEpargne.Depot(montantRetrait);
                                        laNouvelleBalance = ClientCourant.compteEpargne.Balance;
                                        TransacReussi = true;
                                    }
                                }
                            }
                        }
                    }
                }

                if (fondInsufisant)
                {
                    MessageBox.Show("Fond insufisant - Transaction annulé");
                    break;
                    this.Close();
                }
            }
            while (!TransacReussi);
